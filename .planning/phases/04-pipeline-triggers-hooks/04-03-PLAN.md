**4. Frontmatter parser function** (reuse the regex from smoke tests):
```javascript
function parseFrontmatter(content) {
  const match = content.match(/^---\r?\n([\s\S]*?)\r?\n---/);
  if (!match) return null;
  const fm = {};
  const lines = match[1].split(/\r?\n/);
  let currentKey = null;
  let inArray = false;
  for (const line of lines) {
    if (line.match(/^\s*-\s+/) && inArray && currentKey) {
      fm[currentKey].push(line.replace(/^\s*-\s+/, '').trim());
    } else if (line.match(/^(\w[\w-]*):\s*$/)) {
      currentKey = line.match(/^(\w[\w-]*):/)[1];
      fm[currentKey] = [];
      inArray = true;
    } else if (line.match(/^(\w[\w-]*):\s+(.+)/)) {
      const m = line.match(/^(\w[\w-]*):\s+(.*)/);
      fm[m[1]] = m[2].trim();
      currentKey = m[1];
      inArray = false;
    } else if (line.match(/^(\w[\w-]*):\s*>-?\s*$/)) {
      const m = line.match(/^(\w[\w-]*):/);
      fm[m[1]] = '';
      currentKey = m[1];
      inArray = false;
    }
  }
  return fm;
}
```
